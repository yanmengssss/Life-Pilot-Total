### 6.1.2 Nginx 反向代理配置

系统采用 Nginx 作为反向代理服务器，负责将外部请求分发到各个容器服务。Nginx 配置通过服务器上的 nginx-manager 管理界面进行配置，无需手动编写配置文件。

**Nginx Manager 配置要点**

在 nginx-manager 管理界面中添加以下反向代理规则：

| 服务 | 外部端口 | 目标容器 | 目标端口 | 特殊配置 |
|------|---------|---------|---------|----------|
| 前端 | 80/443 | lifepilot-frontend | 3000 | SSL 终端 |
| API | 4000 | lifepilot-server | 4000 | SSE 不缓冲 |
| AI 服务 | 8000 | ai-server | 8000 | 大文件超时 |

**SSE 流式输出配置**

对于 AI 对话的 SSE 流式输出，需要关闭 Nginx 缓冲以确保实时性：

```nginx
# SSE 不缓冲配置（通过 nginx-manager 界面配置）
proxy_buffering off;
proxy_cache off;
X-Accel-Buffering: no;
```

**API 代理配置**

REST API 请求需要传递用户标识和保持长连接：

```nginx
# 代理头配置
proxy_set_header Host $host;
proxy_set_header X-Real-IP $remote_addr;
proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
proxy_http_version 1.1;
proxy_set_header Connection '';
```

**大文件上传配置**

文档上传和视频录制需要较长的超时时间和较大的 body 限制：

```nginx
client_max_body_size 100M;
proxy_read_timeout 300s;
proxy_connect_timeout 75s;
```

### 6.2.2 任务管理模块

**app/api/tasks/route.ts**

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'
import { addToReminderQueue } from '@/lib/redis'

// 获取任务列表
export async function GET(req: NextRequest) {
  const appKey = req.headers.get('x-app-key')
  const { searchParams } = new URL(req.url)
  const status = searchParams.get('status')

  const tasks = await prisma.task.findMany({
    where: {
      appKey,
      ...(status && { status: status as any })
    },
    include: {
      tags: true
    },
    orderBy: [
      { priority: 'asc' },
      { endAt: 'asc' }
    ]
  })

  return NextResponse.json({ tasks })
}

// 创建任务
export async function POST(req: NextRequest) {
  const appKey = req.headers.get('x-app-key')
  const body = await req.json()

  const {
    title,
    content,
    startAt,
    endAt,
    priority = 2,
    tipsFrequency,
    tagIds = []
  } = body

  const task = await prisma.task.create({
    data: {
      title,
      content,
      startAt: new Date(startAt),
      endAt: new Date(endAt),
      priority,
      tipsFrequency,
      appKey: appKey!,
      tags: {
        connect: tagIds.map((id: string) => ({ id }))
      }
    },
    include: {
      tags: true
    }
  })

  // 设置提醒
  if (tipsFrequency && task.endAt) {
    await addToReminderQueue({
      taskId: task.id,
      userId: appKey!,
      remindAt: task.endAt,
      frequency: tipsFrequency
    })
  }

  return NextResponse.json({ task }, { status: 201 })
}

// 更新任务
export async function PATCH(req: NextRequest) {
  const appKey = req.headers.get('x-app-key')
  const { searchParams } = new URL(req.url)
  const taskId = searchParams.get('id')

  if (!taskId) {
    return NextResponse.json({ error: 'Task ID required' }, { status: 400 })
  }

  const body = await req.json()

  const task = await prisma.task.update({
    where: {
      id: taskId,
      appKey
    },
    data: {
      title: body.title,
      content: body.content,
      status: body.status,
      priority: body.priority,
      endAt: body.endAt ? new Date(body.endAt) : undefined,
      tipsFrequency: body.tipsFrequency
    },
    include: {
      tags: true
    }
  })

  return NextResponse.json({ task })
}

// 删除任务
export async function DELETE(req: NextRequest) {
  const appKey = req.headers.get('x-app-key')
  const { searchParams } = new URL(req.url)
  const taskId = searchParams.get('id')

  if (!taskId) {
    return NextResponse.json({ error: 'Task ID required' }, { status: 400 })
  }

  await prisma.task.delete({
    where: {
      id: taskId,
      appKey
    }
  })

  return NextResponse.json({ success: true })
}
```

**hooks/useTaskStore.ts**

```typescript
import { create } from 'zustand'
import { persist } from 'zustand/middleware'

interface Task {
  id: string
  title: string
  content: string
  startAt: string
  endAt: string
  status: 'pending' | 'in_progress' | 'completed'
  priority: 1 | 2 | 3
  tags: Tag[]
  tipsFrequency?: string
}

interface Tag {
  id: string
  name: string
  color: string
}

interface TaskState {
  tasks: Task[]
  loading: boolean
  error: string | null

  fetchTasks: () => Promise<void>
  createTask: (task: Omit<Task, 'id'>) => Promise<void>
  updateTask: (id: string, updates: Partial<Task>) => Promise<void>
  deleteTask: (id: string) => Promise<void>
}

export const useTaskStore = create<TaskState>()(
  persist(
    (set, get) => ({
      tasks: [],
      loading: false,
      error: null,

      fetchTasks: async () => {
        set({ loading: true, error: null })
        try {
          const res = await fetch('/api/tasks', {
            headers: { 'x-app-key': getAppKey() }
          })
          const data = await res.json()
          set({ tasks: data.tasks, loading: false })
        } catch (error) {
          set({ error: 'Failed to fetch tasks', loading: false })
        }
      },

      createTask: async (task) => {
        const res = await fetch('/api/tasks', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'x-app-key': getAppKey()
          },
          body: JSON.stringify(task)
        })
        const data = await res.json()
        set((state) => ({
          tasks: [...state.tasks, data.task]
        }))
      },

      updateTask: async (id, updates) => {
        await fetch(`/api/tasks?id=${id}`, {
          method: 'PATCH',
          headers: {
            'Content-Type': 'application/json',
            'x-app-key': getAppKey()
          },
          body: JSON.stringify(updates)
        })
        set((state) => ({
          tasks: state.tasks.map((t) =>
            t.id === id ? { ...t, ...updates } : t
          )
        }))
      },

      deleteTask: async (id) => {
        await fetch(`/api/tasks?id=${id}`, {
          method: 'DELETE',
          headers: { 'x-app-key': getAppKey() }
        })
        set((state) => ({
          tasks: state.tasks.filter((t) => t.id !== id)
        }))
      }
    }),
    {
      name: 'task-storage',
      partialize: (state) => ({ tasks: state.tasks })
    }
  )
)

function getAppKey() {
  if (typeof window === 'undefined') return ''
  return localStorage.getItem('app-key') || ''
}
```

### 6.2.3 AI 对话模块

**hooks/useChatStream.ts**

```typescript
import { useState, useCallback, useRef, useEffect } from 'react'

interface Message {
  role: 'user' | 'assistant'
  content: string
}

export function useChatStream() {
  const [messages, setMessages] = useState<Message[]>([])
  const [isStreaming, setIsStreaming] = useState(false)
  const [currentContent, setCurrentContent] = useState('')
  const eventSourceRef = useRef<EventSource | null>(null)

  const sendMessage = useCallback(async (content: string) => {
    // 添加用户消息
    setMessages((prev) => [...prev, { role: 'user', content }])
    setIsStreaming(true)
    setCurrentContent('')

    // 创建 SSE 连接
    const url = `/api/chat/stream?message=${encodeURIComponent(content)}`
    const eventSource = new EventSource(url)
    eventSourceRef.current = eventSource

    eventSource.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data)

        switch (data.type) {
          case 'content':
            setCurrentContent((prev) => prev + data.content)
            break
          case 'done':
            setMessages((prev) => [
              ...prev,
              { role: 'assistant', content: currentContent + data.content }
            ])
            setCurrentContent('')
            setIsStreaming(false)
            eventSource.close()
            break
          case 'error':
            console.error('Stream error:', data.message)
            setIsStreaming(false)
            eventSource.close()
            break
        }
      } catch (e) {
        console.error('Failed to parse SSE data:', e)
      }
    }

    eventSource.onerror = () => {
      console.error('EventSource error')
      setIsStreaming(false)
      eventSource.close()
    }
  }, [currentContent])

  const disconnect = useCallback(() => {
    eventSourceRef.current?.close()
    setIsStreaming(false)
  }, [])

  // 清理
  useEffect(() => {
    return () => {
      eventSourceRef.current?.close()
    }
  }, [])

  return { messages, isStreaming, currentContent, sendMessage, disconnect }
}
```

### 6.2.4 视频录制模块

**components/VideoRecorder/CanvasRenderer.ts**

```typescript
interface Sticker {
  id: string
  type: 'emoji' | 'text' | 'image'
  content: string
  x: number
  y: number
  scale: number
  rotation: number
}

export class CanvasRenderer {
  private canvas: HTMLCanvasElement
  private ctx: CanvasRenderingContext2D
  private video: HTMLVideoElement
  private stickers: Sticker[] = []
  private animationId: number = 0

  constructor(canvas: HTMLCanvasElement, video: HTMLVideoElement) {
    this.canvas = canvas
    this.ctx = canvas.getContext('2d')!
    this.video = video
    this.resize()
    window.addEventListener('resize', () => this.resize())
  }

  resize() {
    this.canvas.width = this.video.videoWidth || 1920
    this.canvas.height = this.video.videoHeight || 1080
  }

  addSticker(sticker: Sticker) {
    this.stickers.push(sticker)
  }

  removeSticker(id: string) {
    this.stickers = this.stickers.filter((s) => s.id !== id)
  }

  startRenderLoop() {
    const render = () => {
      this.clear()
      this.drawVideoFrame()
      this.drawStickers()
      this.animationId = requestAnimationFrame(render)
    }
    requestAnimationFrame(render)
  }

  stopRenderLoop() {
    cancelAnimationFrame(this.animationId)
  }

  private clear() {
    this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height)
  }

  private drawVideoFrame() {
    if (this.video.readyState >= 2) {
      this.ctx.drawImage(this.video, 0, 0)
    }
  }

  private drawStickers() {
    for (const sticker of this.stickers) {
      const x = sticker.x * this.canvas.width
      const y = sticker.y * this.canvas.height

      this.ctx.save()
      this.ctx.translate(x, y)
      this.ctx.scale(sticker.scale, sticker.scale)
      this.ctx.rotate((sticker.rotation * Math.PI) / 180)

      switch (sticker.type) {
        case 'emoji':
          this.ctx.font = '64px serif'
          this.ctx.fillText(sticker.content, 0, 0)
          break
        case 'text':
          this.ctx.font = 'bold 32px sans-serif'
          this.ctx.fillStyle = 'white'
          this.ctx.strokeStyle = 'black'
          this.ctx.lineWidth = 2
          this.ctx.strokeText(sticker.content, 0, 0)
          this.ctx.fillText(sticker.content, 0, 0)
          break
      }

      this.ctx.restore()
    }
  }

  getStream(): MediaStream {
    return this.canvas.captureStream(60)
  }
}
```

```typescript
import { StateGraph, Interrupt } from '@langchain/langgraph'
import { z } from 'zod'

// 状态类型
interface WorkflowState {
  messages: Message[]
  intent: string | null
  plan: TaskPlan | null
  inspectionResult: { passed: boolean; reason?: string } | null
  userFeedback: 'accepted' | 'rejected' | null
}

interface Message {
  role: 'user' | 'assistant'
  content: string
}

interface TaskPlan {
  tasks: {
    title: string
    description: string
    startAt: string
    endAt: string
    priority?: number
  }[]
}

// 节点实现
async function routerNode(state: WorkflowState): Promise<Partial<WorkflowState>> {
  const lastMessage = state.messages[state.messages.length - 1].content

  const response = await deepseek.chat.completions.create({
    model: 'deepseek-chat',
    messages: [
      {
        role: 'system',
        content: `判断用户意图：build_todo(创建任务), what_to_do(查询任务), rag(知识问答), travel(出行规划), general(一般对话)`
      },
      { role: 'user', content: lastMessage }
    ]
  })

  const result = JSON.parse(response.choices[0].message.content)
  return { intent: result.intent }
}

async function executorNode(state: WorkflowState): Promise<Partial<WorkflowState>> {
  const lastMessage = state.messages[state.messages.length - 1].content

  const response = await gpt4o.chat.completions.create({
    model: 'gpt-4o',
    messages: [
      {
        role: 'system',
        content: `根据用户输入生成任务计划JSON：{"tasks":[{"title":"","description":"","startAt":"","endAt":"","priority":2}]}`
      },
      { role: 'user', content: lastMessage }
    ]
  })

  const plan = JSON.parse(response.choices[0].message.content)
  return { plan }
}

async function inspectorNode(state: WorkflowState): Promise<Partial<WorkflowState>> {
  const { plan } = state

  const response = await gpt4o.chat.completions.create({
    model: 'gpt-4o',
    messages: [
      {
        role: 'system',
        content: `检查任务计划是否完整合理，返回JSON：{"passed":true/false,"reason":"如果不通过说明原因"}`
      },
      { role: 'user', content: JSON.stringify(plan) }
    ]
  })

  const result = JSON.parse(response.choices[0].message.content)
  return { inspectionResult: result }
}

async function managerNode(state: WorkflowState): Promise<Partial<WorkflowState>> {
  const { plan } = state

  const formatted = plan!.tasks
    .map((t, i) => `${i + 1}. ${t.title}\n   ${t.description}\n   ${t.startAt} - ${t.endAt}`)
    .join('\n\n')

  const message = `计划如下，请确认：\n\n${formatted}`

  return {
    messages: [...state.messages, { role: 'assistant', content: message }]
  }
}

// 构建工作流
const workflow = new StateGraph(WorkflowState)
  .addNode('router', routerNode)
  .addNode('executor', executorNode)
  .addNode('inspector', inspectorNode)
  .addNode('manager', managerNode)
  .addNode('wait_for_confirmation', async (state) => {
    await Interrupt('等待确认')
    return state
  })
  .addEdge('__start__', 'router')
  .addConditionalEdges('router', (state) => state.intent, {
    build_todo: 'executor',
    rag: 'rag_node',
    travel: 'react_node',
    general: 'manager'
  })
  .addEdge('executor', 'inspector')
  .addConditionalEdges('inspector', (state) => {
    return state.inspectionResult?.passed ? 'manager' : 'executor'
  })
  .addEdge('manager', 'wait_for_confirmation')

export const app = workflow.compile()
```

### 6.3.4 RAG 多路召回

**ai-server/services/retriever.py**

```python
import asyncio
from typing import List, Tuple
from qdrant_client import QdrantClient
from neo4j import GraphDatabase
import openai

class MultiPathRetriever:
    def __init__(self):
        self.qdrant = QdrantClient(host='qdrant', port=6333)
        self.neo4j = GraphDatabase.driver(
            'bolt://neo4j:7687',
            auth=('neo4j', 'password')
        )
        self.collection_name = 'documents'

    async def retrieve(
        self,
        query: str,
        user_id: str,
        top_k: int = 20,
        final_k: int = 5
    ) -> List[dict]:
        # 并行三路检索
        dense, sparse, graph = await asyncio.gather(
            self.dense_search(query, user_id, top_k),
            self.sparse_search(query, user_id, top_k),
            self.graph_search(query, user_id, top_k)
        )

        # 合并去重
        merged = self.merge_results(dense, sparse, graph)

        # Cross-Encoder 重排
        reranked = await self.rerank(query, merged, final_k)

        return reranked

    async def dense_search(
        self,
        query: str,
        user_id: str,
        limit: int
    ) -> List[dict]:
        # 获取查询向量
        embedding = openai.Embedding.create(
            model='text-embedding-3-small',
            input=query
        )['data'][0]['embedding']

        # Qdrant 检索
        results = self.qdrant.search(
            collection_name=self.collection_name,
            vector=embedding,
            query_filter={
                'must': [
                    {'key': 'user_id', 'match': {'value': user_id}}
                ]
            },
            limit=limit
        )

        return [
            {'text': r.payload['text'], 'score': r.score, 'source': 'dense'}
            for r in results
        ]

    async def sparse_search(
        self,
        query: str,
        user_id: str,
        limit: int
    ) -> List[dict]:
        # BM25 检索（使用 Elasticsearch）
        # 这里简化处理，实际应连接 ES 集群
        return [
            {'text': f'Sparse result for {query}', 'score': 0.8, 'source': 'sparse'}
        ]

    async def graph_search(
        self,
        query: str,
        user_id: str,
        limit: int
    ) -> List[dict]:
        # 抽取查询实体
        entities = await self.extract_entities(query)

        if not entities:
            return []

        # Cypher 查询
        cypher = """
        MATCH (e:Entity)-[r]-(related)
        WHERE e.name IN $entities AND e.user_id = $userId
        RETURN related.name AS name, type(r) AS relation
        LIMIT $limit
        """

        async with self.neo4j.session() as session:
            result = await session.run(cypher, entities=entities, userId=user_id, limit=limit)
            records = await result.data()

        return [
            {'text': f"{r['name']} --{r['relation']}-->", 'score': 0.9, 'source': 'graph'}
            for r in records
        ]

    async def extract_entities(self, query: str) -> List[str]:
        # LLM 抽取实体
        response = openai.ChatCompletion.create(
            model='gpt-4o-mini',
            messages=[{
                'role': 'user',
                'content': f'从查询中抽取实体名，返回JSON数组："{query}"'
            }]
        )
        return json.loads(response.choices[0].message.content)

    def merge_results(self, dense, sparse, graph) -> List[dict]:
        seen = set()
        merged = []

        all_results = sorted(
            dense + sparse + graph,
            key=lambda x: x['score'],
            reverse=True
        )

        for item in all_results:
            text_hash = hash(item['text']) % 10000
            if text_hash not in seen:
                seen.add(text_hash)
                merged.append(item)

        return merged[:20]

    async def rerank(self, query: str, candidates: List[dict], top_k: int) -> List[dict]:
        # Cross-Encoder 重排
        # 实际使用 SiliconFlow 的 BGE-M3 模型
        pairs = [(query, c['text']) for c in candidates]

        # 调用重排 API
        response = openai.ChatCompletion.create(
            model='gpt-4o',
            messages=[{
                'role': 'user',
                'content': f'根据查询对候选答案评分，返回JSON：{pairs}'
            }]
        )

        scores = json.loads(response.choices[0].message.content)

        for i, c in enumerate(candidates):
            c['score'] = scores[i]

        return sorted(candidates, key=lambda x: x['score'], reverse=True)[:top_k]
```

### 6.3.5 提醒调度服务

**services/reminderScheduler.ts**

```typescript
import Redis from 'ioredis'
import { PrismaClient } from '@prisma/client'

const prisma = new PrismaClient()
const redis = new Redis(process.env.REDIS_URL)

export class ReminderScheduler {
  private isRunning: boolean = false

  async scheduleReminder(
    taskId: string,
    userId: string,
    remindAt: Date,
    frequency?: string
  ): Promise<void> {
    const key = `reminders:${userId}`
    const score = remindAt.getTime()

    await redis.zadd(key, score, JSON.stringify({ taskId, frequency }))
    console.log(`Scheduled reminder for ${taskId} at ${remindAt}`)
  }

  async startPolling(): Promise<void> {
    this.isRunning = true
    console.log('Reminder scheduler started')

    while (this.isRunning) {
      await this.processDueReminders()
      await this.sleep(5000) // 每 5 秒轮询
    }
  }

  stopPolling(): void {
    this.isRunning = false
  }

  private async processDueReminders(): Promise<void> {
    const now = Date.now()
    const keys = await redis.keys('reminders:*')

    for (const key of keys) {
      const userId = key.split(':')[1]

      // 原子性获取并删除到期任务
      const items = await redis.zrangebyscore(key, '-inf', now)

      for (const item of items) {
        const removed = await redis.zrem(key, item)

        if (removed === 1) {
          const { taskId, frequency } = JSON.parse(item)
          await this.sendReminder(userId, taskId)

          // 如果是重复提醒，设置下一次
          if (frequency) {
            const nextTime = this.calculateNextTime(frequency)
            if (nextTime) {
              await this.scheduleReminder(taskId, userId, nextTime, frequency)
            }
          }
        }
      }
    }
  }

  private async sendReminder(userId: string, taskId: string): Promise<void> {
    const task = await prisma.task.findUnique({
      where: { id: taskId, appKey: userId }
    })

    if (!task) return

    // 发送 WebSocket 通知
    // 存储到数据库
    await prisma.notification.create({
      data: {
        userId,
        type: 'task_reminder',
        content: `任务「${task.title}」即将到期`,
        taskId: task.id
      }
    })

    console.log(`Sent reminder for task ${taskId} to user ${userId}`)
  }

  private calculateNextTime(frequency: string): Date | null {
    const now = Date.now()

    switch (frequency) {
      case '1h':
        return new Date(now + 3600000)
      case '1d':
        return new Date(now + 86400000)
      case '3d':
        return new Date(now + 259200000)
      default:
        return null
    }
  }

  private sleep(ms: number): Promise<void> {
    return new Promise((resolve) => setTimeout(resolve, ms))
  }
}

// 启动调度器
const scheduler = new ReminderScheduler()
scheduler.startPolling().catch(console.error)
```

```typescript
import { NextRequest } from 'next/server'
import { generateChatResponse } from '@/services/llm'

export async function GET(req: NextRequest) {
  const encoder = new TextEncoder()
  const appKey = req.headers.get('x-app-key')

              send({ type: 'done' })
              controller.close()
            },
            onError: (error: Error) => {
              send({ type: 'error', message: error.message })
              controller.close()
            }
          }
        )
      } catch (error) {
        send({ type: 'error', message: 'Stream error' })
        controller.close()
      }
    }
  })

      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive',
```
